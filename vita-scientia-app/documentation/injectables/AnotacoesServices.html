import { HttpClient, HttpHeaders } from '@angular/common/http';
import { Injectable } from '@angular/core';
import { PaginaNotas } from '../interface/Paginas/PaginaNotas';
import { Nota } from '../interface/Objetos/Nota';
import * as moment from 'moment';

/**
 * Service responsavel pelas requisições para o back end
 * @author Lucas Alexandre
 */
@Injectable({
  providedIn: 'root',
})
export class AnotacoesServices {
  private url = 'http://localhost:8080/Anotacao';

  get_auth_data: any = JSON.parse(localStorage.getItem('AUTH_COOKIE'));

  globalHeader() {
    return new HttpHeaders({
      Authorization: `Bearer ${this.get_auth_data.token}`,
    });
  }

  constructor(private http: HttpClient) {}

  listaDeAnotacoes(pagina: number) {
    return this.http.get<PaginaNotas>(
      this.url + `/ListaDeNotas/${this.get_auth_data.id}?page=${pagina}`,
      {
        headers: this.globalHeader(),
      }
    );
  }

  pesquisaPeloTitulo(titulo: string, pagina: number) {
    return this.http.get<PaginaNotas>(
      this.url +
        `/PesquisaPeloTitulo/${this.get_auth_data.id}/${titulo}?page=${pagina}`,
      {
        headers: this.globalHeader(),
      }
    );
  }

  pesquisaPeloId(nota_id: number) {
    return this.http.get<PaginaNotas>(
      this.url + `/PesquisaPeloId/${this.get_auth_data.id}/${nota_id}`,
      {
        headers: this.globalHeader(),
      }
    );
  }

  deletaAnotacao(id: number) {
    return this.http.delete(this.url + `/DeletaAnotacao/${id}`, {
      headers: this.globalHeader(),
    });
  }

  novaAnotacao(nota: Nota) {
    nota.data_criacao = moment().format('DD/MM/YYYY HH:mm:ss');
    nota.fk_usuario = this.get_auth_data.id;
    return this.http.post<Nota>(this.url + '/NovaAnotacao', nota, {
      headers: this.globalHeader(),
    });
  }

  atualizaAnotacao(nota: Nota) {
    return this.http.put(this.url + '/AtualizaAnotacao', nota, {
      headers: this.globalHeader(),
    });
  }

  //-------------------------------Pesquisa-------------------------------------------

  notasNaoAssociadosPesquisa(id_pesquisa: number) {
    return this.http.get<[Nota]>(
      this.url +
        `/Pesquisa/AnotacoesNaoAssociadas/${id_pesquisa}/${this.get_auth_data.id}`,
      {
        headers: this.globalHeader(),
      }
    );
  }

  associarAnotacoesPesquisa(id_pesquisa: number, listaDeNotas: number[]) {
    return this.http.post<[Nota]>(
      this.url + `/Pesquisa/AssociarAnotacoes/${id_pesquisa}`,
      listaDeNotas,
      {
        headers: this.globalHeader(),
      }
    );
  }

  desassociarAnotacoesPesquisa(id_pesquisa: number, id_nota: number) {
    return this.http.delete(
      this.url + `/Pesquisa/DesassociarAnotacoes/${id_pesquisa}/${id_nota}`,
      {
        headers: this.globalHeader(),
      }
    );
  }

  //-------------------------------Projeto-------------------------------------------

  notasNaoAssociadosProjeto(id_pesquisa: number) {
    return this.http.get<Nota[]>(
      this.url +
        `/Projeto/AnotacoesNaoAssociadas/${id_pesquisa}/${this.get_auth_data.id}`,
      {
        headers: this.globalHeader(),
      }
    );
  }

  associarAnotacoesProjeto(id_pesquisa: number, listaDeNotas: number[]) {
    return this.http.post<[Nota]>(
      this.url + `/Projeto/AssociarAnotacoes/${id_pesquisa}`,
      listaDeNotas,
      {
        headers: this.globalHeader(),
      }
    );
  }

  desassociarAnotacoesProjeto(id_pesquisa: number, id_nota: number) {
    return this.http.delete(
      this.url + `/Projeto/DesassociarAnotacoes/${id_pesquisa}/${id_nota}`,
      {
        headers: this.globalHeader(),
      }
    );
  }
}
