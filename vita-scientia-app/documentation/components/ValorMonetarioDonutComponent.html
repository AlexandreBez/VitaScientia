import { Component, Input, OnInit, ViewChild } from '@angular/core';
import { ChartOptions } from '../Chart.type';
import { CurrencyPipe } from '@angular/common';
import { ChartComponent } from 'ng-apexcharts';

@Component({
  selector: 'app-valor-monetario-donut',
  templateUrl: './valor-monetario-donut.component.html',
  styleUrls: ['./valor-monetario-donut.component.scss'],
})
export class ValorMonetarioDonutComponent implements OnInit {
  @Input({ required: true }) labels: any[] = [];
  @Input({ required: true }) dados: any[] = [];
  @ViewChild('graficoMonetario') graficoMonetario: ChartComponent;
  public chartOptionsRequisitos: Partial<ChartOptions>;

  constructor(private currencyPipe: CurrencyPipe) {}

  ngOnInit() {
    this.chartOptionsRequisitos = {
      series: this.dados,
      chart: {
        width: '100%',
        height: 380,
        type: 'donut',
      },
      labels: this.labels,
      responsive: [
        {
          breakpoint: 480,
          options: {
            legend: {
              position: 'bottom',
            },
          },
        },
      ],
      dataLabels: {
        enabled: true,
        style: {
          colors: ['#FFF'], // Cor da legenda
        },
      },
      tooltip: {
        enabled: true,
        y: {
          formatter: (value: number) =>
            this.currencyPipe.transform(value, 'BRL', 'symbol', '1.2-2'),
        },
      },
      legend: {
        labels: {
          colors: '#FF6100', // Cor do texto da legenda
        },
      },
    };
  }
}
