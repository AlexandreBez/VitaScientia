import { Component, OnInit, ViewChild } from '@angular/core';
import { ActivatedRoute } from '@angular/router';

import {
  ApexNonAxisChartSeries,
  ApexResponsive,
  ApexChart,
  ApexDataLabels,
  ApexTooltip,
  ApexLegend,
  ChartComponent,
} from 'ng-apexcharts';
import { Projeto } from 'src/app/global/interface/Objetos/Projeto';
import { ProjetoService } from 'src/app/global/services/projeto.service';


export type ChartOptions = {
  series: ApexNonAxisChartSeries;
  chart: ApexChart;
  responsive: ApexResponsive[];
  labels: any;
  dataLabels: ApexDataLabels;
  tooltip: ApexTooltip;
  legend: ApexLegend;
};

@Component({
  selector: 'app-view-projeto',
  templateUrl: './view-projeto.component.html',
  styleUrls: ['./view-projeto.component.scss'],
})
export class ViewProjetoComponent implements OnInit {

  carregando: boolean = false;
  conteudoProjeto: Projeto;
  mensagemDeErro: string = null;
  mensagemDeSucesso: string = null;
  id_projeto: number;
  notaView: any = [];
  descricaoPesquisa: any = [];

  @ViewChild('graficoPesquisa') graficoPesquisa: ChartComponent;
  public chartOptionsProjetos: Partial<ChartOptions>;

  constructor(
    private projetoService: ProjetoService,
    private activatedRoute: ActivatedRoute
  ) {}

  contaQuantidadeStatusDosProjetos(dados: string[]) {
    let aberto = 0;
    let em_espera_verba = 0;
    let em_espera_material = 0;
    let em_espera_outros = 0;
    let finalizado = 0;
    let outros = 0;

    for (let dado of dados) {
      switch (dado) {
        case 'Aberto':
          aberto++;
          break;
        case 'Em espera - Verba':
          em_espera_verba++;
          break;
        case 'Em espera - Material':
          em_espera_material++;
          break;
        case 'Em espera - Outra pesquisa/projeto':
          em_espera_outros++;
          break;
        case 'Finalizado':
          finalizado++;
          break;
        default:
          outros++;
          break;
      }
    }

    return [aberto, em_espera_verba, em_espera_material, em_espera_outros, finalizado, outros];
  }

  ngOnInit() {
    this.activatedRoute.paramMap.subscribe((params) => {
      this.id_projeto = parseInt(params.get('id_projeto'));
      this.viewProjetoId();
    });
  }

  viewProjetoId(){
    this.mensagemDeErro = null;
    this.mensagemDeSucesso = null;
    this.carregando = true;

    this.projetoService.viewProjetoId(this.id_projeto).subscribe(
      (data) => {
        let statusProjetoDados: string[] = [];
        for (let dadosProjeto of data.pesquisa) {
          statusProjetoDados.push(dadosProjeto.status);
        }
        let dadosProjetoSerie = this.contaQuantidadeStatusDosProjetos(statusProjetoDados);
        this.chartOptionsProjetos = {
          series: dadosProjetoSerie,
          chart: {
            width: '100%',
            height: 380,
            type: 'donut',
          },
          labels: ["Aberto", "Em espera - Verba", "Em espera - Material", "Em espera - Outra pesquisa/projeto", "Finalizado", "Outros"],
          responsive: [
            {
              breakpoint: 480,
              options: {
                legend: {
                  position: 'bottom',
                },
              },
            },
          ],
          dataLabels: {
            enabled: true,
            style: {
              colors: ['#FFF'], // Cor da legenda
            },
          },
          tooltip: {
            enabled: true,
          },
          legend: {
            labels: {
              colors: '#FF6100', // Cor do texto da legenda
            },
          },
        };
        this.conteudoProjeto = data;
        this.carregando = false;
      },
      (error) => {
        this.carregando = false;
        this.conteudoProjeto = null;
        this.mensagemDeErro = 'Oops... Algo de errado aconteceu';
      }
    );
  }

}
